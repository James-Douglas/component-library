# Loading with stops
<Story name="Loading with stops">
  {() => {
    const [loadVar, isLoadVar] = useState(false);
    setTimeout(() => {
        isLoadVar(true);
      },
     3000);
    const optionalProps = {
      messages: [
        'Checking your details',
        'Finding great transaction accounts',
        'Sorting by lowest fees',
        'Finding great transaction accounts',
        'Sorting by lowest fees',
      ],
      delayLoading: true
    };
   return (
    <>
    {!loadVar &&  <Loading {...optionalProps}  onLoaded={() => {}} forceStop={loadVar} isDelayMessages />}
    {loadVar &&  <div style={{fontSize: '30px'}}>Loading done</div>}
      </>
    )
  }}
</Story>
