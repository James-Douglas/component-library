# Modal

A modal appended to to the document body that can accept other components as children. State is managed in the parent component, and passed back to the modal.
This allows the Modal component to be used with or without an overlay component.

**Switch to canvas view to see examples**

<Story name="modal" parameters={{ docs: { disable: true } }}>
  {() => {
    const [ visible, setVisible ] = useState(false);
    const modalOpen = () => {
      setVisible(!visible);
    }
    const modalClose = () => {
      setVisible(false);
    }
    return (
      <>
        {visible && <Overlay opacityLevel={0.3} onClose={modalClose} />}
        <Button id='preview-btn' variant='secondary' handleClick={modalOpen}>
          Modal Preview
        </Button>
        <Modal id='preview-modal' visible={visible} handleClose={modalClose} size="sm">
          <p>Some modal content!</p>
        </Modal>
      </>
    )
  }}
</Story>
