<Story name="textarea with props" decorators={[withKnobs]} parameters={{ docs: { disable: true } }}>
  {() => {
    const componentPropsId = 'Component Props';
    const tooltipGroupId = 'Tooltip Props';
    const id = text('id', 'input-one', componentPropsId);
    const name = text('name', 'input-one', componentPropsId);
    const label = text('label', 'Label', componentPropsId);
    const placeholder = text('placeholder', 'Placeholder Text', componentPropsId);
    const textareaValue = text('value', '', componentPropsId);
    const bordered = boolean('bordered', true, componentPropsId);
    const disabled = boolean('disabled', false, componentPropsId);
    const required = boolean('required', false, componentPropsId);
    const prefillValue = text('prefillValue', '', componentPropsId);
    const rows = text('rows', '6', componentPropsId); // can be set via the css width property
    const wrap = select('wrap', ['hard', 'soft'], 'soft', componentPropsId);
    const readonly = boolean('readonly', false, componentPropsId);
    const maxLength = text('maxlength', '', componentPropsId);
    const maxChars = text('maxchars', '500', componentPropsId);
    const validationMessage = text('validationMessage', '', componentPropsId);
    const { title, body, placement, screenReaderLabel, variant } = getTootipKnobs(tooltipGroupId);
    return (
      <TextArea
        tooltip={{
          title,
          body,
          placement,
          screenReaderLabel,
          variant
        }}
        id={id}
        name={name}
        label={label}
        placeholder={placeholder}
        value={textareaValue}
        bordered={bordered}
        disabled={disabled}
        required={required}
        prefillValue={prefillValue}
        rows={rows}
        wrap={wrap}
        readonly={readonly}
        maxChars={maxChars}
        maxLength={maxLength}
        validationMessage={validationMessage}
      />
    );
  }}
</Story>
