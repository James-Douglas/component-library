# Typography

The Typography component can be used to render text content. A `variant` is required to determine what style text should be rendered.

<Preview>
  <Story name="typography">
    <DemoContainer>
      <Typography variant="body1">Typography component</Typography>
    </DemoContainer>
  </Story>
</Preview>

<Story name="typography with props" decorators={[withKnobs]} parameters={{ docs: { disable: true } }}>
  {() => {
    const variant = select('variant', ['h1','h2','h3','h4','h5','h6','subtitle1','subtitle2','body1','body2','caption','overline','srOnly'], 'body1');
    const component = text('component', null);
    const align = select('align', ['inherit', 'left', 'center', 'right', 'justify'], 'inherit');
    const color = select('color', ['initial', 'inherit'], 'inherit');
    const display = select('display', ['initial', 'block', 'inline'], 'block');
    const noWrap = boolean('noWrap', false);
    const noMargins = boolean('noMargins', true);
    return (
      <DemoContainer>
        <Typography
          variant={variant}
          component={component}
          align={align}
          color={color}
          display={display}
          noWrap={noWrap}
          noMargins={noMargins}
      >
        Here is some text
      </Typography>
      </DemoContainer>
    );
  }}
</Story>
