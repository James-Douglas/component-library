## ResponsiveLayout

The ResponsiveLayout provides a means to render children elements that are within a range of breakpoint values.

<Preview>
  <Story name="responsive layout">
    <div>
      <ResponsiveLayout isMobile>
        <Typography variant="body1">Renders on mobile devices</Typography>
      </ResponsiveLayout>
      <ResponsiveLayout isDesktop>
        <Typography variant="body1">Renders on desktop devices</Typography>
      </ResponsiveLayout>
      <ResponsiveLayout is="md">
        <Typography variant="body1">Renders on devices that is at md breakpoint</Typography>
      </ResponsiveLayout>
      <ResponsiveLayout is="lg">
        <Typography variant="body1">Renders on devices that is at lg breakpoint</Typography>
      </ResponsiveLayout>
      <ResponsiveLayout isFrom="sm">
        <Typography variant="body1">Renders on devices that are from sm breakpoint</Typography>
      </ResponsiveLayout>
      <ResponsiveLayout isFrom="lg">
        <Typography variant="body1">Renders on devices that are from lg breakpoint</Typography>
      </ResponsiveLayout>
      <ResponsiveLayout upTo="sm">
        <Typography variant="body1">Renders on devices that are up to sm breakpoint</Typography>
      </ResponsiveLayout>
      <ResponsiveLayout isFrom="sm" upTo="lg">
        <Typography variant="body1">Renders on devices within sm and lg</Typography>
      </ResponsiveLayout>
      <ResponsiveLayout isFrom="sm" upTo="lg" andIs={false}>
        <Typography variant="body1">Doesn't render when andIs prop is false</Typography>
      </ResponsiveLayout>
    </div>
  </Story>
</Preview>

## CustomResponsiveLayout

The CustomResponsiveLayout provides a means to render children elements that are within a range of custom breakpoint values.

<Preview>
  <Story name="custom responsive layout">
    {() => {
      const customBreakpoints = {
        extraSmall: 300,
        small: 600,
        medium: 900,
        large: 1200,
        extraLarge: 1600,
      };
      return (
        <div>
          <CustomBreakpointsProvider customBreakpoints={customBreakpoints}>
            <CustomResponsiveLayout is="extraSmall">
              <Typography variant="body1">Renders on devices that is at extraSmall breakpoint</Typography>
            </CustomResponsiveLayout>
            <CustomResponsiveLayout is="medium">
              <Typography variant="body1">Renders on devices that is at medium breakpoint</Typography>
            </CustomResponsiveLayout>
            <CustomResponsiveLayout isFrom="small">
              <Typography variant="body1">Renders on devices that are from small breakpoint</Typography>
            </CustomResponsiveLayout>
            <CustomResponsiveLayout isFrom="large">
              <Typography variant="body1">Renders on devices that are from large breakpoint</Typography>
            </CustomResponsiveLayout>
            <CustomResponsiveLayout upTo="small">
              <Typography variant="body1">Renders on devices that are up to small breakpoint</Typography>
            </CustomResponsiveLayout>
            <CustomResponsiveLayout upTo="large">
              <Typography variant="body1">Renders on devices that are up to large breakpoint</Typography>
            </CustomResponsiveLayout>
            <CustomResponsiveLayout isFrom="small" upTo="large">
              <Typography variant="body1">Renders on devices that are between small and large breakpoint</Typography>
            </CustomResponsiveLayout>
            <CustomResponsiveLayout isFrom="small" upTo="large" andIs={false}>
              <Typography variant="body1">Doesn't render when andIs prop is false</Typography>
            </CustomResponsiveLayout>
          </CustomBreakpointsProvider>
        </div>
      )
    }}
  </Story>
</Preview>
