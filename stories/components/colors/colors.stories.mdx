# CTM Color Guidelines

CtM products adapt intelligently and flexibly to cater to people's preferences. This includes complying with AA standard
contrast ratios. To do this, choose primary, secondary, and extended colours that support usability by ensuring
sufficient colour contrast between elements so that people with low vision can see and use the interface.

## Primary Color Palette

<Story name="Primary Color Palette">
  {() => {
    const { colors } = ctmTheme;
    let primaryColors = [];
    for (let [key, value] of Object.entries(colors)) {
      if (key.includes('primary')) {
        primaryColors.push({
          name: key,
          color: value
        })
      }
    }
    return (
      <Container>
        <Row>
          {primaryColors.map(({name, color}) => {
            return (
            <Column md={4} lg={3} sm={6} key={`primary_${name}`}>
              <ColorDisplay name={name} color={color} />
            </Column>
            )
          })}
        </Row>
       <h4>For usage please view the <a href='https://zeroheight.com/9942937b5/p/943375-colors'>CTM Design System</a> documentation. </h4>
      </Container>
    )
  }}
</Story>

## Base Color Palette

<Story name="Base Color Palette">
  {() => {
    const { colors } = ctmTheme;
    let blackWhiteColors = [];
    let greyColors = [];
    for (let [key, value] of Object.entries(colors)) {
      if (key.includes('black') || key.includes('white')) {
        blackWhiteColors.push({name: key, color: value})
      }
      if (key.includes('grey')) {
        greyColors.push({name: key, color: value})
      }
    }
    return (
      <>
        <Container>
          <Row><h3>Black &amp; White</h3></Row>
          <Row>
            {blackWhiteColors.map(({name, color}) => (
              <Column md={4} lg={3} sm={6} key={`blackWhite_${name}`}>
                <ColorDisplay name={name} color={color} />
              </Column>
            ))}
          </Row>
        </Container>
        <Container>
          <Row><h3>Base / Grey</h3></Row>
          <Row>
            {greyColors.map(({name, color}) => (
              <Column md={4} lg={3} sm={6} key={`grey_${name}`}>
                <ColorDisplay name={name} color={color} />
              </Column>
            ))}
          </Row>
          <h4>For usage please view the <a href='https://zeroheight.com/9942937b5/p/943375-colors'>CTM Design System</a> documentation. </h4>
        </Container>
      </>
    )
  }}
</Story>

## Functional Color Palette

<Story name="Functional Color Palette">
  {() => {
    const { colors } = ctmTheme;
    let successColors = [];
    let warningColors = [];
    let errorColors = [];
    for (let [key, value] of Object.entries(colors)) {
      if (key.includes('success')) {
        successColors.push({name: key, color: value});
      }
      if (key.includes('warning')) {
        warningColors.push({name: key, color: value})
      }
      if (key.includes('error')) {
        errorColors.push({name: key, color: value});
      }
    }
    return (
      <>
        <Container>
          <Row><h3>Success</h3></Row>
          <Row>
            {successColors.map(({name, color}) => (
              <Column md={4} lg={3} sm={6}  key={`success_${name}`}>
                <ColorDisplay name={name} color={color} />
              </Column>
            ))}
          </Row>
        </Container>
        <Container>
          <Row><h3>Warning</h3></Row>
          <Row>
            {warningColors.map(({name, color}) => (
              <Column md={4} lg={3} sm={6}  key={`warning_${name}`}>
                <ColorDisplay name={name} color={color} />
              </Column>
            ))}
          </Row>
        </Container>
        <Container>
          <Row><h3>Error</h3></Row>
          <Row>
            {errorColors.map(({name, color}) => (
              <Column md={4} lg={3} sm={6} key={`error_${name}`}>
                <ColorDisplay name={name} color={color} />
              </Column>
            ))}
          </Row>
          <h4>For usage please view the <a href='https://zeroheight.com/9942937b5/p/943375-colors'>CTM Design System</a> documentation. </h4>
        </Container>
      </>
    )
  }}
</Story>
